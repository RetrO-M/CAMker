import requests

class CamKer:
    def __init__(self):
        self.headers = {
            "User-Agent":       "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/98.0.4758.80 Safari/537.36 Edg/98.0.1108.62",
            "Accept-Language":  "en-US,en;q=0.9"
        }
        self.countries = {
            "FRANCE":          "FR/camera.txt",
            "USA":             "US/camera.txt",
            "RUSSIA":          "RU/camera.txt",
            "NETHERLANDS":     "NL/camera.txt",
            "CANADA":          "CA/camera.txt",
            "ITALY":           "IT/camera.txt",
            "GERMANY":         "DE/camera.txt",
            "POLAND":          "PL/camera.txt",
            "SWEDEN":          "SE/camera.txt"
        }
        self.settings = {
            "timeout": 10,
            "allow_redirects": False,
            "proxies": None
        }
    def start():
        try:
            self = CamKer()
            for country in self.countries:
                url = f'https://RetrO-M.github.io/CAMker/{self.countries[country]}'
                response = requests.get(
                    url, 
                    headers=self.headers, 
                    timeout=self.settings["timeout"],
                    allow_redirects=self.settings["allow_redirects"],
                    proxies=self.settings["proxies"]
                )
                if response.status_code == 200:
                    print(response.text)
                elif response.status_code == 404:
                    print(f'\033[0;91m[!] \033[0;97mAn error occurred at "\033[0;91m{url}\033[0;97m"')
        except requests.RequestException as e:
            print(f"\033[0;91m[!] \033[0;97mNetwork error : {e}")
        
if __name__ == '__main__':
    CamKer.start()
